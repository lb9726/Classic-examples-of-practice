{

    QJsonDocument doucment = QJsonDocument::fromJson(byteArray, &jsonError);  // 转化为 JSON 文档
    if (!doucment.isNull() && (jsonError.error ==QJsonParseError::NoError))   // 解析未发生错误
    {
        return true;
    }
    else
    {
        return false;
